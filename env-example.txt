# Xtream Codes Normalizing Proxy - Optional Settings
# 
# NO CONFIGURATION REQUIRED for basic usage!
# Just run: docker-compose up -d
#
# Provider credentials are configured in Dispatcharr, not here.
# See README.md for details.

# =====================================================
# Optional: Server Settings
# =====================================================

# Port the proxy listens on (default: 8765)
# PORT=8765

# How long to cache API responses, in hours (default: 6)
# CACHE_HOURS=6

# =====================================================
# Optional: Stream Naming Settings
# =====================================================

# Enable/disable channel name normalization
# When true (default): Normalizes channel names (removes prefixes, cleans up)
# When false: Passes through channel names unchanged from provider
#
# NORMALIZE_NAMES=true

# Groups that should NOT have (provider) (group) suffixes added.
# Useful for groups with "Auto Channel Sync" enabled in Dispatcharr,
# where stream names become channel names directly.
#
# Format: comma-separated EXACT group names (case-insensitive)
# Copy the exact group names from Dispatcharr's Groups settings
#
# NO_SUFFIX_GROUPS=NL| ODIDO VERMAAK RAW GOLD,NL| 8TKO PPV,NL| MAX PPV,NL| VIAPLAY PPV

# =====================================================
# Optional: Smart Validation (for Redirect mode)
# =====================================================

# Enable smart validation: test provider streams before redirecting
# This allows Dispatcharr's failover to work with Redirect mode.
# When a stream is dead (405, timeout), proxy returns 502 instead of 302,
# so Dispatcharr tries the next available stream.
#
# Default: true (enabled)
# SMART_VALIDATION=true

# Timeout for stream validation in seconds (default: 5)
# Increase if your provider is slow to respond
# VALIDATION_TIMEOUT=5

# =====================================================
# Optional: Video Proxying
# =====================================================

# Enable video proxying through this server
# When false (default): Returns 302 redirect, video goes directly to provider
# When true: Actually streams video through this proxy (uses bandwidth!)
#
# Use cases:
# - false: Direct streaming, saves bandwidth, but client must follow redirects
# - true: All traffic through proxy, works with any client, but uses bandwidth
#
# Default: false (redirect mode)
# PROXY_VIDEO=false

# =====================================================
# Optional: Direct URL Injection
# =====================================================

# Inject direct provider URLs in API responses
# When enabled, adds 'direct_source' field to each stream pointing
# directly to the IPTV provider instead of through the proxy.
# This allows clients that support it to bypass the proxy for playback.
#
# Default: true (enabled)
# DIRECT_URLS=true
